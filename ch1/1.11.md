(define (f n)
  (cond ((= n 0) 0)
        ((= n 1) 1)
        ((= n 2) 2)
        (else (+ (f (- n 1))
                 (* 2 (f (- n 2)))
                 (* 3 (f (- n 3)))))))

(define (f n)
  (define (improve a b c)
    (+ (* 3 a) (* 2 b) c))
  (define (f-iter fn-3 fn-2 fn-1 count)
    (define improved (improve fn-3 fn-2 fn-1))
    (cond ((= count 3) improved)
          (else (f-iter fn-2 fn-1 improved (- count 1)))))
  (cond ((= n 0) 0)
        ((= n 1) 1)
        ((= n 2) 2)
        (else (f-iter 0 1 2 n))))
